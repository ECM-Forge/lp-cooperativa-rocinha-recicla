(function(){
  const track    = document.getElementById('tcarousel-track');
  const dotsWrap = document.getElementById('tcarousel-dots');
  const btnPrev  = document.getElementById('tcarousel-prev');
  const btnNext  = document.getElementById('tcarousel-next');
  if(!track) return;

  const total = track.children.length;
  let page = 0;

  function perView(){
    if(window.innerWidth <= 560) return 1;
    if(window.innerWidth <= 860) return 2;
    return 3;
  }

  // Calcula o total de pûÀginas avanûÏando de 1 em 1 card para evitar slots vazios
  function totalPages(){
    const pv = perView();
    return Math.max(1, total - pv + 1);
  }

  function buildDots(){
    dotsWrap.innerHTML = '';
    const n = totalPages();
    for(let i = 0; i < n; i++){
      const d = document.createElement('button');
      d.className = 'tcarousel__dot';
      d.setAttribute('aria-label', 'PûÀgina ' + (i+1));
      d.addEventListener('click', () => goTo(i));
      dotsWrap.appendChild(d);
    }
  }

  function goTo(p){
    const pv   = perView();
    const pages = totalPages();
    page = Math.max(0, Math.min(p, pages - 1));

    const gap   = 24;
    const cardW = (track.parentElement.parentElement.offsetWidth - gap * (pv - 1)) / pv;
    
    // Desloca de 1 em 1 card (multiplicando apenas por page, em vez de page * pv)
    track.style.transform = `translateX(-${page * (cardW + gap)}px)`;

    dotsWrap.querySelectorAll('.tcarousel__dot').forEach((d, i) =>
      d.classList.toggle('is-active', i === page)
    );

    btnPrev.disabled = page === 0;
    btnNext.disabled = page >= pages - 1;
  }

  btnPrev.addEventListener('click', () => goTo(page - 1));
  btnNext.addEventListener('click', () => goTo(page + 1));

  function startTimer(){
    return setInterval(() => goTo(page >= totalPages() - 1 ? 0 : page + 1), 5000);
  }

  let timer = startTimer();
  const carousel = track.closest('.tcarousel');
  carousel.addEventListener('mouseenter', () => clearInterval(timer));
  carousel.addEventListener('mouseleave', () => { timer = startTimer(); });

  window.addEventListener('resize', () => { buildDots(); goTo(0); });

  buildDots();
  goTo(0);
})();
</script>
